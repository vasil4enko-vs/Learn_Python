# Ответ к задаче 5.2.2

```python
p = [0] * 10
count = 0

while count < 5:
    index = int(input())
    if p[index] == 1:
        continue

    p[index] = 1
    count += 1

print(*p)
```

```python
p = [0] * 10
while p.count(1) != 5:
    d = int(input())

        continue
    p[d] = 1
print(*p)
```

p = [0] * 10

```python
p = [0] * 10 #создаем список из 10 нулей
count = 0 #заводим счетчик единиц
while count < 5: #заводим цикл while с условием "пока счетчик меньше 5"
    i = int(input()) #вводим индекс элемента списка p
    if p[i] != 0: #если элемент с введенным индексом не равен нулю пропускаем одну итерацию цикла
        continue
    else:
        p[i] = 1 #иначе (т.е. элемент с введенным индексом равен 0) заменяем этот элемент на 1
        count += 1 #увеличиваем счетчик на единицу
print(*p) #после завершения цикла выводим получившийся список на печать (используем * чтобы убрать [])
```
