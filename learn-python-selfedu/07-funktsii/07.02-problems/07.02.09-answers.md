```python
def minim_x_maxim(minim: int, maxim: int) -> int:
    """Возвращает произведение минимального и максимального чисел из списка digs."""
    return minim * maxim


digs = [int(x) for x in map(int, input().split())]

print(minim_x_maxim(min(digs), max(digs)))
```
