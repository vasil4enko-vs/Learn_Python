```python
def wrap_in_tag(text: str, tag: str = "h1") -> str:
    """Оборачивает текст в HTML-тег.

    Args:
        text(str): Текст для оборачивания.
        tag(str): HTML-тег.

    Returns:
        str: Текст обёрнутый в HTML-тег.

    """
    return f"<{tag}>{text}</{tag}>"


text = input()

print(wrap_in_tag(text))
print(wrap_in_tag(text, "div"))
```
