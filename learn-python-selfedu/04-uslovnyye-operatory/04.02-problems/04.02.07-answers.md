# Ответ к задаче 4.2.7

```python
DAYS_WEEK = ["понедельник", "вторник", "среда", "четверг", "пятница", "суббота", "воскресенье"]

k = int(input())

if 1 <= k <= 365:
    print(DAYS_WEEK[k % 7 -1])
else:
    print("Введены ошибочные данные.")
```

```python
d = int(input())
d = d % 7
if d == 1:
    print('понедельник')
elif d == 2:
    print('вторник')
elif d == 3:
    print('среда')
elif d == 4:
    print('четверг')
elif d == 5:
    print('пятница')
elif d == 6:
    print('суббота')
elif d == 0:
    print('воскресенье')
```

```python
k = int(input())
days = list('понедельник, вторник, среда, четверг, пятница, суббота, воскресенье'.split(', '))
if 1 <= k <= 365:
    print(days[k % 7 - 1])
else:
    print(f'число {k} не соответствует условиям')
```
